294  private boolean checkFields() {
295  try {
296  Date today = dateFormat.parse(dateFormat.format(new Date())); //In questo modo ho solo la parte di data odierna e non l'orario che crea problemi con il controllo successivo
297  if (txt_destinazione.getText().isBlank()) {
298  showWarningDialog("La destinazione non pu essere vuota");
299  return false;
300  } else if (txtf_peso.getText().isBlank() || weightFormat.parse(txtf_peso.getText()).doubleValue() < 0.001) {
301  showWarningDialog("Il peso non pu essere vuoto o di valore inferiore o uguale a 0");
302  return false;
303  } else if (txtf_data.getText().isBlank() || dateFormat.parse(txtf_data.getText()).before(today)) {
304  showWarningDialog("La data non pu essere vuota o antecedente a quella odierna");
305  return false;
306  } else if (chk_spedizioneAssicurata.isSelected() && (txtf_valoreAssicurato.getText().isEmpty() || valueFormat.parse(txtf_valoreAssicurato.getText()).doubleValue() < 0.01)) {
307  showWarningDialog("Una spedizione assicurata non pu avere un valore assicurato vuoto di valore inferiore o uguale a 0");
308  return false;
309  }
310  }catch(Exception e) {
311  showWarningDialog("Errore durante il controllo dei dati");
312  return false;
313  }
